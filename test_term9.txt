let zipWith f xs ys =
      case xs of
        Nil -> Nil []
        Cons x xs' ->
          case ys of
            Nil -> Nil []
            Cons y ys' -> Cons [f x y, zipWith f xs' ys']
    ix l i = case l of
      Cons x xs -> if i == 0 then x else ix xs (i - 1)
in Left [ix (zipWith (\x y -> x + y) (Cons [1, Cons [2, Nil []]]) (Cons [3, Cons [4, Nil []]])) 1]
